%% Hopfield classification
clear, clc, format compact
 
% CONFIGURATION
p1 = [1 -1 1]; % Type A prototype
p2 = [-1 -1 -1]; % Type B prototype
p3 = [-1 1 -1]; % parcel to classify
 
% Weights matrix
W = [1.2 0 0; 0 0.2 0; 0 0 1.2]; % 3x3 matrix
 
% Bias vector
b = [0; -0.9; 0]; % 3x1 column vector
 
% CALCULATION
a = p3'; % initializing the neuron with the input vector
 
% REPEAT CALCULATION
i = 1;
while any(a ~= p1) & any(a ~= p2) % repeat while any output vector is not equal to Type A or Type B patterns
old_a = a;
a = satlins(W*a+b); % output
a = round(a,2); % round a to two decimal places
i = i + 1;
end
 
% CLASSIFICATION
if (a == p1)
type = 'Type A';
else
type = 'Type B';
end
 
% PRINT RESULTS
g=sprintf('%d ', p3); % formats data in array p3 into string
fprintf('The input parcel p:[ %s] is of %s.\n',g,type);
