function v = viewProtypeVect(v)
figure
% Creating the axes in the figure
ax = axes(); % default Cartesian axes
xlim(ax, [-1 1]); % range of x-axis
ylim(ax, [-1 1]); % range of y-axis
zlim(ax, [-1 1]); % range of z-axis
ax.LineWidth = 2; % changing the thickness of the axes
view(v) % camera line of sight
title('Prototype Vectors (Input Space)')
xlabel('p1 (size)','Color','r');
ylabel('p2 (shape)','Color','r');
zlabel('p3 (weight)','Color','r');
grid on
hold on
[q1,q2] = meshgrid(-1:.1:1); % range of values
x=zeros(21); % zeros matrix 21x21
plot3(x,q1,q2,'color',[0.2 0.8 0.8])
plot3(x,q2,q1,'color',[0.2 0.8 0.8])
hold on
 
% Customizing axes to origin crossover
hAxis = gca;
hAxis.XRuler.FirstCrossoverValue = 0; % X crossover with Y axis
hAxis.XRuler.SecondCrossoverValue = 0; % X crossover with Z axis
hAxis.YRuler.FirstCrossoverValue = 0; % Y crossover with X axis
hAxis.YRuler.SecondCrossoverValue = 0; % Y crossover with Z axis
hAxis.ZRuler.FirstCrossoverValue = 0; % Z crossover with X axis
hAxis.ZRuler.SecondCrossoverValue = 0; % Z crossover with Y axis
hAxis.YColor = [1 1 0]; % yellow
hAxis.XColor = 'k'; % black
hAxis.ZColor = 'b'; % blue
hAxis.FontSize = 10;
hAxis.FontWeight = 'bold';
hAxis.Color = [0.4 0.4 0.4]; % background color is grey
hAxis.GridColor = 'y';
hAxis.GridAlpha = 0.2;
% Plotting the vectors
XSource = [0, 0];
YSource = [0, 0];
ZSource = [0, 0];
USource = [1, -1];
VSource = [-1,-1];
WSource = [1, -1];
quiver3(XSource, YSource, ZSource, USource, VSource, WSource,'linewidth',3, 'color','r');
text(USource, VSource, WSource,['Type A'; 'Type B'], 'Color', 'r', 'FontWeight', 'bold')
hold off
end
