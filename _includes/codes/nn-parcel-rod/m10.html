clear, clc, format compact
% Neuron input
p1 = [1 -1 1]; % Type A prototype
p2 = [-1 -1 -1]; % Type B prototype
p3 = [-1 1 -1]; % not-so-perfect Type B
p = [p1;p2;p3];
 
% Neuron weights
W = [1 0 0]; % single row weights matrix
 
% Neuron bias
b = 0;
 
for i = 1:length(p)
net_input = p(i,:)*W'+b;
neuron_output = hardlims(net_input);
if i == 1
type = 'Type A';
elseif i == 2
type = 'Type B';
else
type = 'not-so-perfect Type B';
end
fprintf('For an input of %s the neuron output is %d.\n',type, neuron_output);
fprintf('');
end
