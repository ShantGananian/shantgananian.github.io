% Plotting four activation functions for a simple neuron
clear, clc, format compact
 
% Defining the neuron parameters
p =-4:0.001:4; % range and resolution of input values
n =2*p + 3; % net input
 
% The functions
z1 = tansig(n); % Hyperbolic tangent sigmoid transfer function
z2 = 2./(1+exp(-n)) - 1; % Exponential transfer function
z3 = hardlim(n); % Hard-limit transfer function
z4 = purelin(n); % Linear transfer function
 
% Plotting
figure(figure)
plot(p,z1,'b',p,z2,'r',p,z3,'g',p,z4,'y','LineWidth',2) % plot the functions
hold on
x_at_y0 = interp1(n,p,0);
y_at_x0 = interp1(z1,z4,0);
plot([x_at_y0 x_at_y0],get(gca,'ylim'),'k--') % plot a vertical line at the threshold value
plot(get(gca,'xlim'),[y_at_x0 y_at_x0],'k--') % plot a horizontal line at the threshold value
plot(x_at_y0, 0, 'k*', 'MarkerSize',15) % mark the threshold point
plot(get(gca,'xlim'),[1 1],'r--') % plot horizontal line at a = 1
plot(get(gca,'xlim'),[-1 -1],'r--') % plot horizontal line at a = -1
legend('Tansig','Exp','Hard-limit','Linear')
set(gca, 'ylim', [-4 4]);
set(gca, 'ytick', -4:1:4);
title('Activation Functions')
xlabel('Input')
ylabel('Neuron output')
grid on
hold off
