% Getting a sample
random_index = randi(length(force))
sample = force(random_index,:);
% Plotting the sample
figure
plot(force','c')
grid on, hold on
plot(sample,'b')
xlabel('Time')
ylabel('Force')
% Predicting the quality
q = net(sample');
% Labeling network output
q = double(q > threshold)'
% Commenting the network response
if q==target(random_index)
title(sprintf('Quality: %d (correct network response)',q))
else
title(sprintf('Quality: %d (wrong network response)',q))
end
